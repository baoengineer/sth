import { theme } from '@/src/stitches';
import { colors } from '@/src/variables';
import { ColorPreview } from '@/app/components';

### Scale

<div>
	{Object.values(colors.scales).map((token) => {
		const colorName = token.replace('$', '');
		const colorObject = theme.colors[colorName];
		return (
			<ColorPreview
				key={colorObject.token}
				color={colorObject}
			/>
		);
	})}
</div>
